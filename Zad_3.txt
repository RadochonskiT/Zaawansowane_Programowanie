def IsEven(number: int) -> bool:
    restFromDividing = number % 2
    isEven = True
    if (restFromDividing > 0):
        isEven = False
    return isEven

isEven = IsEven(12)
if (isEven):
    print("Liczba parzysta")
else:
    print("Liczba nieparzysta")

def PrintEven(numbers):
    for number in numbers:
        if number % 2 == 0:
            print(number)

numbers = range(10, 21, 1)
PrintEven(numbers)
